# Runtime dependencies for the Collision Engineers OCR host (Linux container,
# Azure Functions on Azure Container Apps). Installed into the image by the
# Dockerfile `pip install`. Unlike the FC1 parser, this host runs as a CONTAINER,
# so it CAN carry the tesseract OS binary (added in the Dockerfile via apt-get) —
# that one binary is the reason this host uses a custom container.

# Azure Functions Python worker API.
azure-functions>=1.21.0

# --- Document OCR (scanned/image-only PDFs) ---------------------------------
PyMuPDF>=1.24.0        # licensed/approved; render PDF pages -> images (import: fitz)
Pillow>=10.0.0         # image handling for the rendered pages
pytesseract>=0.3.10    # thin WRAPPER over the tesseract binary (provided by the image)

# --- Registration-plate OCR (overview photo) --------------------------------
# fast-alpr bundles a detector (YOLO) + plate OCR (CCT), runs on ONNX Runtime,
# CPU, MIT-licensed. Pinned conservatively; CPU wheels only (no GPU).
fast-alpr>=0.1.0
onnxruntime>=1.17.0
numpy>=1.26.0

# --- Document Intelligence Read fallback (OCR_PROVIDER=docintel) -------------
# We call DI Read's REST API directly (async analyze -> poll) with `requests`,
# rather than the azure-ai-documentintelligence SDK, to keep the image lean and
# the call shape explicit. Endpoint+key come from app settings / Key Vault ref.
requests>=2.31.0

# --- Vendored parser engine (cedocumentmapper_v2/, full 12-field extraction) --
# Only reached when the engine is present in the image (_engine_available());
# these are its actual runtime deps beyond PyMuPDF/Pillow/pytesseract above.
# pywin32 is deliberately NOT listed: it's Windows-only (this container is
# Linux) and the engine's own doc.py already falls back cleanly without it.
pypdf[image]>=5.0.0
python-docx>=1.1.0
extract-msg>=0.45.0
jsonschema>=4.22.0
olefile>=0.47
